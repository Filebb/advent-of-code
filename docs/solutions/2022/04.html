input <- readLines(con = "inputs/2022/04.txt")

    separate_wider_delim(col = original,
                         delim = ",",
                         names = c("elf1", "elf2")) |>

                         delim = "-",

                         delim = "-",
                         names = c("elf2_min", "elf2_max")) |>

               (elf1_min >= elf2_min & elf1_max <= elf2_max) |

           )
