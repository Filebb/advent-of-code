input <- read_lines(file = "inputs/2015/01.txt")

moves <- tibble(input  = str_split_1(string = input, pattern = ""),
                move   = case_match(input, "(" ~  1L, ")" ~ -1L),
                height = cumsum(move))

moves |> 
    tail(1) |> 
    pull(height) |> 
    cat()
