library(tidyverse)

input <- read_lines(file = "inputs/2022/01.txt") |> as.integer()

elf_calories <- tibble(calories = input,
                       elf = cumsum(is.na(input)) + 1L) |> 
    filter(!is.na(calories))

elf_calories |> 
    group_by(elf) |> 
    summarise(total_calories = sum(calories)) |> 
    slice_max(total_calories) |> 
    pull(total_calories) |> 
    cat()
