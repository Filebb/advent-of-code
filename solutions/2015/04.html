# number of attempts, this is empirically enough
n <- 2.55e5

md5_hashes <- tibble(number = 1:n,
                     string = paste0(input, number),
                     hash   = map_chr(.x = string, .f = \(x) md5(x, serialize = FALSE)))

md5_hashes |> 
    filter(str_sub(string = hash, start = 1, end = 5) == "00000") |> 
    pull(number) |> 
    cat()
## 254575
