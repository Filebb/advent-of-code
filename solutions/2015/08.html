strings <- tibble(original = input,
                  code     = str_length(string = input),
                  memory   = map_int(.x = input, 
                                     .f = \(string) parse(text = string) |> 
                                         eval() |> 
                                         nchar(type = "bytes"))) |> 
    mutate(diff = code - memory)

strings |>     
    summarise(diff = sum(diff)) |> 
    pull(diff) |> 
    cat()
